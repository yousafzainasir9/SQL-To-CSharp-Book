Chapter 33

Real-World E-Commerce Project

You get a requirement: "Build an e-commerce API." A SQL developer designs tables and stored procedures. A C# developer builds layers — domain, data access, services, API. This chapter puts every concept from the book together into one production project.

01

Requirement: "Design the Domain Models"

Business Requirement
We need an e-commerce system. Customers place orders. Each order has line items. Products belong to categories. Track prices, quantities, stock levels, and order status.
SQL Developer Thinks Five tables: Customers, Categories, Products, Orders, OrderItems. Foreign keys enforce relationships. DECIMAL for money. INT for stock. CHECK constraints for positive values. UNIQUE on Email and SKU.
C# Developer Thinks Five classes with navigation properties. ICollection<T> for one-to-many. Enum for OrderStatus. Nullable types for optional fields. EF conventions handle the rest.
SQL — Create All Tables
CREATE TABLE Categories (
    Id   INT IDENTITY(1,1) PRIMARY KEY,
    Name NVARCHAR(100) NOT NULL
);

CREATE TABLE Customers (
    Id        INT IDENTITY(1,1) PRIMARY KEY,
    Name      NVARCHAR(100) NOT NULL,
    Email     NVARCHAR(255) NOT NULL UNIQUE,
    IsActive  BIT NOT NULL DEFAULT 1,
    CreatedAt DATETIME2 DEFAULT GETDATE()
);

CREATE TABLE Products (
    Id         INT IDENTITY(1,1) PRIMARY KEY,
    Name       NVARCHAR(200) NOT NULL,
    Sku        NVARCHAR(50) NOT NULL UNIQUE,
    Price      DECIMAL(10,2) NOT NULL
               CHECK (Price >= 0),
    Stock      INT NOT NULL DEFAULT 0,
    CategoryId INT NOT NULL
               REFERENCES Categories(Id),
    IsActive   BIT NOT NULL DEFAULT 1
);

CREATE TABLE Orders (
    Id         INT IDENTITY(1,1) PRIMARY KEY,
    CustomerId INT NOT NULL
               REFERENCES Customers(Id),
    Status     INT NOT NULL DEFAULT 0,
    Total      DECIMAL(12,2) NOT NULL DEFAULT 0,
    CreatedAt  DATETIME2 DEFAULT GETDATE(),
    RowVersion ROWVERSION
);

CREATE TABLE OrderItems (
    Id        INT IDENTITY(1,1) PRIMARY KEY,
    OrderId   INT NOT NULL
              REFERENCES Orders(Id),
    ProductId INT NOT NULL
              REFERENCES Products(Id),
    Quantity  INT NOT NULL CHECK (Quantity > 0),
    UnitPrice DECIMAL(10,2) NOT NULL
);

02

Requirement: "Configure the Database Layer"

Business Requirement
Set up the database context with all tables. Define relationships, constraints, indexes, and defaults using Fluent API. Support concurrency on orders.
SQL Developer Thinks The DDL already defines constraints — NOT NULL, UNIQUE, FOREIGN KEY, CHECK, DEFAULT, ROWVERSION. Migrations should generate matching SQL.
C# Developer Thinks DbContext with DbSet for each entity. OnModelCreating configures everything via Fluent API — property constraints, relationship cascades, indexes, concurrency tokens.
C# — AppDbContext
public class AppDbContext : DbContext
{
    public DbSet<Category> Categories => Set<Category>();
    public DbSet<Customer> Customers => Set<Customer>();
    public DbSet<Product> Products => Set<Product>();
    public DbSet<Order> Orders => Set<Order>();
    public DbSet<OrderItem> OrderItems => Set<OrderItem>();

    public AppDbContext(DbContextOptions<AppDbContext> options)
        : base(options) { }

    protected override void OnModelCreating(ModelBuilder mb)
    {
        // Category
        mb.Entity<Category>(e =>
        {
            e.Property(c => c.Name).IsRequired().HasMaxLength(100);
        });

        // Customer
        mb.Entity<Customer>(e =>
        {
            e.Property(c => c.Name).IsRequired().HasMaxLength(100);
            e.Property(c => c.Email).IsRequired().HasMaxLength(255);
            e.HasIndex(c => c.Email).IsUnique();
            e.Property(c => c.CreatedAt).HasDefaultValueSql("GETDATE()");
        });

        // Product
        mb.Entity<Product>(e =>
        {
            e.Property(p => p.Name).IsRequired().HasMaxLength(200);
            e.Property(p => p.Sku).IsRequired().HasMaxLength(50);
            e.HasIndex(p => p.Sku).IsUnique();
            e.HasCheckConstraint("CK_Product_Price", "[Price] >= 0");
            e.Property(p => p.Stock).HasDefaultValue(0);
            e.HasOne(p => p.Category)
             .WithMany(c => c.Products)
             .HasForeignKey(p => p.CategoryId);
        });

        // Order
        mb.Entity<Order>(e =>
        {
            e.HasOne(o => o.Customer)
             .WithMany(c => c.Orders)
             .HasForeignKey(o => o.CustomerId);
            e.Property(o => o.CreatedAt).HasDefaultValueSql("GETDATE()");
            e.Property(o => o.RowVersion).IsRowVersion();
        });

        // OrderItem
        mb.Entity<OrderItem>(e =>
        {
            e.HasOne(oi => oi.Order)
             .WithMany(o => o.Items)
             .HasForeignKey(oi => oi.OrderId);
            e.HasOne(oi => oi.Product)
             .WithMany(p => p.OrderItems)
             .HasForeignKey(oi => oi.ProductId);
            e.HasCheckConstraint("CK_OrderItem_Qty", "[Quantity] > 0");
        });
    }
}

03

Requirement: "Build the Repository Layer"

Business Requirement
Encapsulate all data access behind repositories. Create a generic repository for common CRUD, and specific repositories for domain-specific queries like "get pending orders" or "get orders by customer."
SQL Developer Thinks This is like stored procedures — each repo method maps to a specific query. GetPendingAsync = SELECT * FROM Orders WHERE Status = 0. GetByCustomerAsync = SELECT with JOIN on OrderItems.
C# Developer Thinks Generic IRepository<T> handles GetById, GetAll, Add, Update, Delete. IOrderRepository extends it with domain queries. All reads use AsNoTracking(). All methods accept CancellationToken.
C# — Generic Repository
public interface IRepository<T> where T : class
{
    Task<T?> GetByIdAsync(int id, CancellationToken ct = default);
    Task<IReadOnlyList<T>> GetAllAsync(CancellationToken ct = default);
    void Add(T entity);
    void Update(T entity);
    void Remove(T entity);
}

public class Repository<T> : IRepository<T> where T : class
{
    protected readonly AppDbContext _db;

    public Repository(AppDbContext db) => _db = db;

    public async Task<T?> GetByIdAsync(int id, CancellationToken ct = default)
        => await _db.Set<T>().FindAsync(new object[] { id }, ct);

    public async Task<IReadOnlyList<T>> GetAllAsync(CancellationToken ct = default)
        => await _db.Set<T>().AsNoTracking().ToListAsync(ct);

    public void Add(T entity) => _db.Set<T>().Add(entity);
    public void Update(T entity) => _db.Set<T>().Update(entity);
    public void Remove(T entity) => _db.Set<T>().Remove(entity);
}
C# — Order-Specific Repository
public interface IOrderRepository : IRepository<Order>
{
    Task<IReadOnlyList<Order>> GetByCustomerAsync(
        int customerId, CancellationToken ct = default);
    Task<IReadOnlyList<Order>> GetPendingAsync(
        CancellationToken ct = default);
    Task<Order?> GetWithItemsAsync(
        int orderId, CancellationToken ct = default);
}

public class OrderRepository : Repository<Order>, IOrderRepository
{
    public OrderRepository(AppDbContext db) : base(db) { }

    public async Task<IReadOnlyList<Order>> GetByCustomerAsync(
        int customerId, CancellationToken ct = default)
        => await _db.Orders.AsNoTracking()
            .Where(o => o.CustomerId == customerId)
            .Include(o => o.Items)
                .ThenInclude(i => i.Product)
            .OrderByDescending(o => o.CreatedAt)
            .ToListAsync(ct);

    public async Task<IReadOnlyList<Order>> GetPendingAsync(
        CancellationToken ct = default)
        => await _db.Orders.AsNoTracking()
            .Where(o => o.Status == OrderStatus.Pending)
            .Include(o => o.Customer)
            .ToListAsync(ct);

    public async Task<Order?> GetWithItemsAsync(
        int orderId, CancellationToken ct = default)
        => await _db.Orders
            .Include(o => o.Items)
                .ThenInclude(i => i.Product)
            .FirstOrDefaultAsync(o => o.Id == orderId, ct);
}
SQL — What GetByCustomerAsync Generates
SELECT o.*, oi.*, p.*
FROM Orders o
INNER JOIN OrderItems oi
    ON oi.OrderId = o.Id
INNER JOIN Products p
    ON p.Id = oi.ProductId
WHERE o.CustomerId = @customerId
ORDER BY o.CreatedAt DESC;
LINQ — Query Syntax Equivalent
var orders = await (
    from o in db.Orders.AsNoTracking()
    where o.CustomerId == customerId
    orderby o.CreatedAt descending
    select o
).Include(o => o.Items)
    .ThenInclude(i => i.Product)
 .ToListAsync(ct);
04

Requirement: "Coordinate Multiple Repositories"

Business Requirement
When creating an order, we update Products (reduce stock), create an Order, and create OrderItems — all in a single transaction. If any step fails, nothing should save.
SQL Developer Thinks BEGIN TRANSACTION → INSERT Order → INSERT OrderItems → UPDATE Products SET Stock = Stock - @qty → COMMIT. One transaction wraps all statements.
C# Developer Thinks Unit of Work pattern. All repos share the same DbContext. Call Add/Update on repos, then one SaveAsync() commits everything in a single transaction.
SQL — Transaction
BEGIN TRANSACTION;

INSERT INTO Orders (CustomerId, Status, Total)
VALUES (@CustomerId, 0, @Total);

DECLARE @OrderId INT = SCOPE_IDENTITY();

INSERT INTO OrderItems
    (OrderId, ProductId, Quantity, UnitPrice)
VALUES
    (@OrderId, @Pid1, @Qty1, @Price1),
    (@OrderId, @Pid2, @Qty2, @Price2);

UPDATE Products
SET Stock = Stock - @Qty1
WHERE Id = @Pid1;

UPDATE Products
SET Stock = Stock - @Qty2
WHERE Id = @Pid2;

COMMIT;
C# — Unit of Work
public interface IUnitOfWork : IDisposable
{
    ICustomerRepository Customers { get; }
    IOrderRepository Orders { get; }
    IRepository<Product> Products { get; }
    IRepository<Category> Categories { get; }
    Task<int> SaveAsync(
        CancellationToken ct = default);
}

public class UnitOfWork : IUnitOfWork
{
    private readonly AppDbContext _db;

    public ICustomerRepository Customers { get; }
    public IOrderRepository Orders { get; }
    public IRepository<Product> Products { get; }
    public IRepository<Category> Categories { get; }

    public UnitOfWork(AppDbContext db)
    {
        _db = db;
        Customers = new CustomerRepository(db);
        Orders = new OrderRepository(db);
        Products = new Repository<Product>(db);
        Categories = new Repository<Category>(db);
    }

    public async Task<int> SaveAsync(
        CancellationToken ct = default)
        => await _db.SaveChangesAsync(ct);

    public void Dispose() => _db.Dispose();
}
05

Requirement: "Create an Order with Validation"

Business Requirement
When a customer places an order: validate the customer exists and is active, check each product has enough stock, calculate the total, reduce stock, save everything atomically. Return a DTO, not the entity.
SQL Developer Thinks A stored procedure with IF EXISTS checks, a cursor or loop for items, stock validation with @@ROWCOUNT, and RAISERROR for failures. Everything in one transaction.
C# Developer Thinks A service class with CreateOrderAsync. Inject IUnitOfWork. Validate, build the Order, iterate items, deduct stock, call SaveAsync. Map to DTO before returning. Handle concurrency with try/catch on DbUpdateConcurrencyException.
C# — DTOs

C# — OrderService
public interface IOrderService
{
    Task<OrderDto> CreateOrderAsync(
        CreateOrderRequest request, CancellationToken ct = default);
    Task<IReadOnlyList<OrderDto>> GetCustomerOrdersAsync(
        int customerId, CancellationToken ct = default);
}

public class OrderService : IOrderService
{
    private readonly IUnitOfWork _uow;

    public OrderService(IUnitOfWork uow) => _uow = uow;

    public async Task<OrderDto> CreateOrderAsync(
        CreateOrderRequest request, CancellationToken ct = default)
    {
        // 1. Validate customer
        var customer = await _uow.Customers
            .GetByIdAsync(request.CustomerId, ct)
            ?? throw new NotFoundException("Customer not found");

        if (!customer.IsActive)
            throw new BadRequestException("Customer is inactive");

        // 2. Build order
        var order = new Order
        {
            CustomerId = customer.Id,
            Status = OrderStatus.Pending,
            Items = new List<OrderItem>()
        };

        // 3. Process each item
        decimal total = 0;
        foreach (var item in request.Items)
        {
            var product = await _uow.Products
                .GetByIdAsync(item.ProductId, ct)
                ?? throw new NotFoundException(
                    $"Product {item.ProductId} not found");

            if (product.Stock < item.Quantity)
                throw new BadRequestException(
                    $"Insufficient stock for {product.Name}");

            order.Items.Add(new OrderItem
            {
                ProductId = product.Id,
                Quantity = item.Quantity,
                UnitPrice = product.Price
            });

            product.Stock -= item.Quantity;
            _uow.Products.Update(product);
            total += product.Price * item.Quantity;
        }

        // 4. Save atomically
        order.Total = total;
        _uow.Orders.Add(order);
        await _uow.SaveAsync(ct);

        // 5. Return DTO
        return MapToDto(order, customer);
    }

    public async Task<IReadOnlyList<OrderDto>> GetCustomerOrdersAsync(
        int customerId, CancellationToken ct = default)
    {
        var orders = await _uow.Orders
            .GetByCustomerAsync(customerId, ct);

        return orders.Select(o => MapToDto(o, o.Customer))
            .ToList();
    }

    private static OrderDto MapToDto(Order o, Customer c) => new(
        o.Id,
        c.Name,
        o.Status,
        o.Total,
        o.CreatedAt,
        o.Items.Select(i => new OrderItemDto(
            i.Product?.Name ?? "Unknown",
            i.Quantity,
            i.UnitPrice,
            i.Quantity * i.UnitPrice
        )).ToList()
    );
}

Requirement: "Expose REST API Endpoints"

Business Requirement
Create API endpoints: POST /api/orders to place an order, GET /api/orders/{id} to fetch order details, GET /api/customers/{id}/orders to list a customer's orders. Return proper HTTP status codes.
SQL Developer Thinks Each endpoint maps to a query: POST triggers the INSERT transaction, GET /orders/{id} is a SELECT with JOIN, GET /customers/{id}/orders is a filtered SELECT. HTTP 201, 200, 404 correspond to INSERT success, SELECT found, SELECT empty.
C# Developer Thinks Thin controller that delegates to IOrderService. Constructor injection. CancellationToken on every action. Return CreatedAtAction for POST, Ok for GET, NotFound when null. Exception middleware handles domain exceptions globally.
C# — API Controller
[ApiController]
[Route("api/orders")]
public class OrdersController : ControllerBase
{
    private readonly IOrderService _orderService;

    public OrdersController(IOrderService orderService)
        => _orderService = orderService;

    [HttpPost]
    public async Task<ActionResult<OrderDto>> CreateOrder(
        CreateOrderRequest request, CancellationToken ct)
    {
        var order = await _orderService.CreateOrderAsync(request, ct);
        return CreatedAtAction(
            nameof(GetOrder),
            new { id = order.Id },
            order);
    }

    [HttpGet("{id:int}")]
    public async Task<ActionResult<OrderDto>> GetOrder(
        int id, CancellationToken ct)
    {
        var orders = await _orderService
            .GetCustomerOrdersAsync(id, ct);
        return Ok(orders);
    }
}

[ApiController]
[Route("api/customers/{customerId:int}/orders")]
public class CustomerOrdersController : ControllerBase
{
    private readonly IOrderService _orderService;

    public CustomerOrdersController(IOrderService orderService)
        => _orderService = orderService;

    [HttpGet]
    public async Task<ActionResult<IReadOnlyList<OrderDto>>> GetOrders(
        int customerId, CancellationToken ct)
    {
        var orders = await _orderService
            .GetCustomerOrdersAsync(customerId, ct);
        return Ok(orders);
    }
}
C# — Global Exception Middleware
public class ExceptionMiddleware
{
    private readonly RequestDelegate _next;

    public ExceptionMiddleware(RequestDelegate next) => _next = next;

    public async Task InvokeAsync(HttpContext context)
    {
        try
        {
            await _next(context);
        }
        catch (NotFoundException ex)
        {
            context.Response.StatusCode = 404;
            await context.Response.WriteAsJsonAsync(
                new { error = ex.Message });
        }
        catch (BadRequestException ex)
        {
            context.Response.StatusCode = 400;
            await context.Response.WriteAsJsonAsync(
                new { error = ex.Message });
        }
        catch (DbUpdateConcurrencyException)
        {
            context.Response.StatusCode = 409;
            await context.Response.WriteAsJsonAsync(
                new { error = "Data was modified by another user" });
        }
    }
}

Requirement: "Build a Dashboard with Dapper"

Business Requirement
The admin dashboard needs fast read queries: total revenue, orders by status, top customers, and top products. These are read-only reports — use Dapper for performance alongside EF Core for writes.
SQL Developer Thinks SUM(Total) for revenue, GROUP BY Status for order counts, TOP 5 with JOIN for top customers and products. These are reporting queries — raw SQL is fastest.
C# Developer Thinks CQRS approach: EF Core for commands (insert, update, delete), Dapper for queries (reports, dashboards). Inject IDbConnection separately from DbContext. Map directly to DTOs.
C# — Dashboard DTOs
public record DashboardDto(
    decimal TotalRevenue,
    int TotalOrders,
    List<OrderStatusCount> OrdersByStatus,
    List<TopCustomerDto> TopCustomers,
    List<TopProductDto> TopProducts);

public record OrderStatusCount(string Status, int Count);
public record TopCustomerDto(string Name, decimal TotalSpent, int OrderCount);
public record TopProductDto(string Name, int TotalSold, decimal Revenue);
SQL — Dashboard Queries
-- Total revenue
SELECT SUM(Total) AS TotalRevenue,
       COUNT(*) AS TotalOrders
FROM Orders
WHERE Status != 4; -- Not cancelled

-- Orders by status
SELECT Status, COUNT(*) AS Count
FROM Orders
GROUP BY Status;

-- Top 5 customers
SELECT TOP 5
    c.Name,
    SUM(o.Total) AS TotalSpent,
    COUNT(o.Id) AS OrderCount
FROM Customers c
INNER JOIN Orders o ON o.CustomerId = c.Id
WHERE o.Status != 4
GROUP BY c.Name
ORDER BY TotalSpent DESC;

-- Top 5 products
SELECT TOP 5
    p.Name,
    SUM(oi.Quantity) AS TotalSold,
    SUM(oi.Quantity * oi.UnitPrice)
        AS Revenue
FROM Products p
INNER JOIN OrderItems oi
    ON oi.ProductId = p.Id
INNER JOIN Orders o
    ON o.Id = oi.OrderId
WHERE o.Status != 4
GROUP BY p.Name
ORDER BY Revenue DESC;
Dapper — Dashboard Service
public class DashboardService
{
    private readonly IDbConnection _conn;

    public DashboardService(
        IDbConnection conn) => _conn = conn;

    public async Task<DashboardDto>
        GetDashboardAsync(
        CancellationToken ct = default)
    {
        const string sql = @"
            SELECT SUM(Total), COUNT(*)
            FROM Orders WHERE Status != 4;

            SELECT Status, COUNT(*) AS Count
            FROM Orders GROUP BY Status;

            SELECT TOP 5 c.Name,
                SUM(o.Total) AS TotalSpent,
                COUNT(o.Id) AS OrderCount
            FROM Customers c
            JOIN Orders o
                ON o.CustomerId = c.Id
            WHERE o.Status != 4
            GROUP BY c.Name
            ORDER BY TotalSpent DESC;

            SELECT TOP 5 p.Name,
                SUM(oi.Quantity) AS TotalSold,
                SUM(oi.Quantity * oi.UnitPrice)
                    AS Revenue
            FROM Products p
            JOIN OrderItems oi
                ON oi.ProductId = p.Id
            JOIN Orders o
                ON o.Id = oi.OrderId
            WHERE o.Status != 4
            GROUP BY p.Name
            ORDER BY Revenue DESC;";

        using var multi = await _conn
            .QueryMultipleAsync(sql);

        var summary = await multi
            .ReadFirstAsync<dynamic>();
        var byStatus = (await multi
            .ReadAsync<OrderStatusCount>())
            .ToList();
        var topCustomers = (await multi
            .ReadAsync<TopCustomerDto>())
            .ToList();
        var topProducts = (await multi
            .ReadAsync<TopProductDto>())
            .ToList();

        return new DashboardDto(
            (decimal)summary.TotalRevenue,
            (int)summary.TotalOrders,
            byStatus,
            topCustomers,
            topProducts);
    }
}
